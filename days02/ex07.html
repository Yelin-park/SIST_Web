package days06; import java.util.Scanner; public class Ex01 { public static void main(String[] args) { char ch; System.out.print("> 하나의 문자를 입력하세요: "); Scanner sc = new Scanner(System.in); String str = sc.next(); ch = str.charAt(0); if(Character.isUpperCase(ch)) { System.out.println("대문자입니다."); } else if(Character.isLowerCase(ch)) { System.out.println("소문자입니다."); } else if('가' <= ch && ch <= '힣') { System.out.println("한글입니다."); } else if(Character.isDigit(ch)) { System.out.println("숫자입니다."); } else if(ch == '#' || ch == '@' || ch == '%' || ch == '!') { System.out.println("특수문자입니다."); } else { System.out.println("모르는 문자입니다."); } } }
